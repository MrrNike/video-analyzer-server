function runAnimation(targetElement, callback, min, max) { let current = 0; let target = Math.floor(Math.random() * (max - min) + min); let interval = setInterval(() => { // Sürət rəqəminin oynaması (realistik görünmək üçün) let jitter = Math.random() * 5; let displayVal = (current + jitter).toFixed(1); mainSpeed.innerText = displayVal; targetElement.innerText = displayVal; // Dairəvi tərəzini hərəkət etdir (Dashoffset 628 tam dövrədir) let offset = 628 - (628 * (current / 100)); progressCircle.style.strokeDashoffset = offset; if (current < target) { current += 1.5; } else { clearInterval(interval); mainSpeed.innerText = target.toFixed(1); targetElement.innerText = target.toFixed(1); callback(target); } }, 50); }
